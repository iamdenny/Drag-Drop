1 jindo.UIComponent = jindo.$Class({
  2 	/** @lends jindo.UIComponent.prototype */
  3 		
  4 	/**
  5 	 * jindo.UIComponent를 초기화한다.
  6 	 * @constructs
  7 	 * @class UI Component에 상속되어 사용되는 Jindo Component의 Core
  8 	 * @extends jindo.Component
  9 	 */
 10 	$init : function() {
 11 		this._bIsActivating = false; //컴포넌트의 활성화 여부
 12 	},
 13 
 14 	/**
 15 	 * 컴포넌트의 활성여부를 가져온다.
 16 	 * @return {Boolean}
 17 	 */
 18 	isActivating : function() {
 19 		return this._bIsActivating;
 20 	},
 21 
 22 	/**
 23 	 * 컴포넌트를 활성화한다.
 24 	 * _onActivate 메소드를 수행하므로 반드시 상속받는 클래스에 _onActivate 메소드가 정의되어야한다.
 25 	 * @return {this}
 26 	 */
 27 	activate : function() {
 28 		if (this.isActivating()) {
 29 			return this;
 30 		}
 31 		this._bIsActivating = true;
 32 		
 33 		if (arguments.length > 0) {
 34 			this._onActivate.apply(this, arguments);
 35 		} else {
 36 			this._onActivate();
 37 		}
 38 				
 39 		return this;
 40 	},
 41 	
 42 	/**
 43 	 * 컴포넌트를 비활성화한다.
 44 	 * _onDeactivate 메소드를 수행하므로 반드시 상속받는 클래스에 _onDeactivate 메소드가 정의되어야한다.
 45 	 * @return {this}
 46 	 */
 47 	deactivate : function() {
 48 		if (!this.isActivating()) {
 49 			return this;
 50 		}
 51 		this._bIsActivating = false;
 52 		
 53 		if (arguments.length > 0) {
 54 			this._onDeactivate.apply(this, arguments);
 55 		} else {
 56 			this._onDeactivate();
 57 		}
 58 		
 59 		return this;
 60 	}
 61 }).extend(jindo.Component);
